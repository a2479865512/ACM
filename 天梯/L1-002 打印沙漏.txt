#include<cstdio>
#include<algorithm>
#include<cmath>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int a, i;
	char c;
	scanf("%d %c", &a, &c);
	for (i = 0; i < 100; i++)
	{
		if (a < i*i * 2 - 1)
		{
			i--;
			break;
		}
	}
	for (int j = i; j > 0; j--)
	{
		int k;
		for (k = 0; k < i - j; k++)
		{
			printf(" ");
		}
		for (k = 2 * j - 1; k > 0; k--)
		{
			printf("%c", c);
		}
		//for( k=0;k<i-j;k++)
		// {
	  	//    printf(" ");
		// }
		printf("\n");
	}
	for (int j = 2; j < i + 1; j++)
	{
		int k;
		for (k = 0; k < i - j; k++)
		{
			printf(" ");
		}
		for (k = 2 * j - 1; k > 0; k--)
		{
			printf("%c", c);
		}
		// for( k=0;k<i-j;k++)
		// {
		//   printf(" ");
		// }
		printf("\n");
	}
	if (a != 0)
		printf("%d\n", a - i * i * 2 + 1);
	return 0;
}

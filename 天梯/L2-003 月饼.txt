#include<cstdio>
#include<algorithm>
#include<cmath>
#include<bits/stdc++.h>
using namespace std;
struct yb
{
	double s;
	double a;
	bool operator<(const yb &x)const
	{
		return x.a*s > a*x.s;
	}

}a[1000];
int main()
{
	int n, k;
	double s = 0;
	cin >> n >> k;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i].a;
	}
	for (int i = 0; i < n; i++)
	{
		cin >> a[i].s;
	}
	sort(a, a + n);
	for (int i = 0; i < n; i++)
	{
		if (k >= a[i].a)
		{
			k -= a[i].a;
			s += a[i].s;
		}
		else {
			s += k * a[i].s / a[i].a;
			break;
		}
	}
	printf("%.2f", s);
	return 0;
}

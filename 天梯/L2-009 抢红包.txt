#include <iostream>
#include<cstdio>
#include<algorithm>
#include <queue>
#include<cmath>
#include<bits/stdc++.h>
using namespace std;

struct A
{
	int num;
	int money;
	int n;
	bool operator<(const A &a)const
	{
		if (a.money != money)
			return money > a.money;
		else if (a.n != n)
			return n > a.n;
		else return num < a.num;
	}
}aa[10004];
int n;
int a[10004], t, k, b;
int c[10004];
int main()
{
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d", &t);
		while (t--)
		{
			scanf("%d %d", &k, &b);
			a[k] += b;
			c[k]++;
			a[i] -= b;
		}
	}
	for (int i = 1; i <= n; i++)
	{
		//aa[i]={i,a[i],c[i]};
		aa[i].num = i;
		aa[i].money = a[i];
		aa[i].n = c[i];
	}
	sort(aa + 1, aa + n + 1);
	for (int i = 1; i <= n; i++)
	{
		printf("%d %.2f\n", aa[i].num, aa[i].money / 100.0);
	}

}

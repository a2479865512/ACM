#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
#define pb push_back
const int MAXN = 1e5 + 5;

struct Edge {
    int u, v, t;
};

int vis[MAXN];
vector <Edge> edge[MAXN];
int ans[MAXN];
int top = 0;

void dfs(int x) {
    //ans[++top] = x;
    for(int i = 0; i < edge[x].size(); i++) {
        int u = edge[x][i].u;
        int v = edge[x][i].v;
        int tt = edge[x][i].t;
        if(!vis[tt]) {
            vis[tt] = 1;
            dfs(v);
            ans[++top] = v;
        }
    }
}

int main() {
    int n ,m;
    int t = 1;
    cin >> n >> m;
    for(int i = 0; i < m; i++) {
        int u, v;
        scanf("%d%d", &u, &v);
        edge[u].pb(Edge{u, v, t++});
        edge[v].pb(Edge{v, u, t++});
    }
    dfs(1);
    cout << "1" << endl;
    for(int i = top; i > 0; i--) cout << ans[i] << endl;
    return 0;
}

#include<cstdio>
#include<cstring>
#include<iostream>
#include<queue>
#include<vector>
#include<cmath>
#include<algorithm>
using namespace std;
const int MAXN = 105;
int line[MAXN][MAXN];//二分连线
int r[MAXN], vis[MAXN];
int k, m, n;
int sum = 0, t = 1, s, s1;
bool found(int x)
{
	for (int i = 1; i <= m; i++)
	{
		if (line[x][i] && !vis[i])
		{
			vis[i] = 1;
			if (r[i] == 0 || found(r[i]))
			{
				r[i] = x;
				return 1;
			}
		}
	}
	return 0;
}
int main()
{
	int x, y;
	while (~scanf("%d %d %d", &n, &m, &k))
	{
		s = sum = 0;
		memset(line, 0, sizeof(line));
		memset(r, 0, sizeof(r));
		for (int i = 0; i < k; i++)
		{
			scanf("%d %d", &x, &y);
			line[x][y] = 1;
		}
		for (int i = 1; i <= n; i++)
		{
			memset(vis, 0, sizeof(vis));
			if (found(i))s++;
		}
		for (int i = 1; i <= n; i++)
			for (int j = 1; j <= m; j++)
				if (line[i][j])
				{
					line[i][j] = 0;
					s1 = 0;
					memset(r, 0, sizeof(r));
					for (int i1 = 1; i1 <= n; i1++)
					{
						memset(vis, 0, sizeof(vis));
						if (found(i1))s1++;
					}
					if (s1 < s)sum++;
					line[i][j] = 1;
				}
		printf("Board %d have %d important blanks for %d chessmen.\n", t++, sum, s);
	}
	return 0;
}

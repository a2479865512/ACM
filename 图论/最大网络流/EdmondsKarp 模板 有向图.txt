#include<queue>
using namespace std;
const int maxn = 205;
const int inf = 0x7fffffff;

int r[maxn][maxn]; //残留网络，初始化为原图
bool vis[maxn];
int pre[maxn];//记录上一节点
int m, n;

bool bfs(int s, int t)  //寻找一条从s到t的增广路，若找到返回true
{
	int p;
	queue<int > q;
	memset(pre, -1, sizeof(pre));
	memset(vis, false, sizeof(vis));

	pre[s] = s;
	vis[s] = true;
	q.push(s);
	while (!q.empty())
	{
		p = q.front();
		q.pop();
		for (int i = 1; i <= n; i++)
		{
			if (r[p][i] > 0 && !vis[i])
			{
				pre[i] = p;
				vis[i] = true;
				if (i == t) return true;
				q.push(i);
			}
		}
	}
	return false;
}

int EdmondsKarp(int s, int t)
{
	int flow = 0, d, i;
	while (bfs(s, t))
	{
		d = inf;
		for (i = t; i != s; i = pre[i])//回溯路径
			d = d < r[pre[i]][i] ? d : r[pre[i]][i];//得到最小残量，即本次更新流量
		for (i = t; i != s; i = pre[i])
		{
			r[pre[i]][i] -= d;//残留网络更新；
			r[i][pre[i]] += d;//返回的负权边；
		}
		flow += d;//最大流更新
	}
	return flow;
}


int main()
{
	while (scanf("%d%d", &m, &n) != EOF)
	{
		int u, v, w;
		memset(r, 0, sizeof(r));///
		for (int i = 0; i < m; i++)
		{
			scanf("%d%d%d", &u, &v, &w);
			r[u][v] += w;//两条管道合成一条边
		}
		printf("%d\n", EdmondsKarp(1, n));
	}
	return 0;
}

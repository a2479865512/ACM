#include <iostream>
#include <cstdio>

using namespace std;
const int maxn=200005;
int fa[maxn],sum[maxn];

int Find(int x)
{
    if(x!=fa[x])
    {
        int f=fa[x];
        fa[x]=Find(f);
        sum[x]+=sum[f];
    }
    return fa[x];
}

int main()
{
    int n,m;
    while(~scanf("%d%d",&n,&m))
    {
        for(int i=0;i<=n;i++)
        {
            fa[i]=i;
            sum[i]=0;
        }
        int ans=0;
        int u,v,w;
        for(int i=0;i<m;i++)
        {
            scanf("%d%d%d",&u,&v,&w);
            u=u-1;
            int r1=Find(u);
            int r2=Find(v);
            if(r1!=r2)
            {
                fa[r2]=r1;
                sum[r2]=sum[u]-sum[v]+w;
            }
            else if(sum[v]-sum[u]!=w)
            {
                    ans++;
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}

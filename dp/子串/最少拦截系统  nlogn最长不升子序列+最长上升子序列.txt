#include <iostream>
#include<cstdio>
#include<algorithm>
#include<bits/stdc++.h>
#include<functional>

using namespace std;
const int MAX = 1e5 + 5;
int f[MAX];//1开始 升序列
int l[MAX];
int a[MAX];//原数
//int pos[MAX];//序列号
int p = 1;//总数
int p2 = 1;
struct cmp { bool operator()(int a, int b) { return a > b; } };
int main()
{
	int i = 2;
	scanf("%d", &a[1]);
	f[1] = l[1] = a[1];
	while (~scanf("%d", &a[i]))
	{
		if (a[i] > f[p])
			f[++p] = a[i];
		else
			f[lower_bound(f + 1, f + p + 1, a[i]) - f] = a[i];
		if (a[i] <= l[p2])//等于可以
			l[++p2] = a[i];
		else
			l[upper_bound(l + 1, l + p2 + 1, a[i], cmp()) - l] = a[i];
		i++;
	}
	printf("%d\n", p2);
	printf("%d\n", p);
	return 0;
}

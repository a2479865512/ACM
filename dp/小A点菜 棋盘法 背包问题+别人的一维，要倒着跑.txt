int main()
{
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d",a+i);
    sort(a+1,a+n+1);
    for(int i=1;i<=n;i++)
    {
            d[i][a[i]]=d[i-1][a[i]]+1;
            for(int j=1;j<=m;j++)
                if(d[i-1][j])
                {
                    if(a[i]+j<=m)
                   d[i][j+a[i]]=d[i-1][j+a[i]]+d[i-1][j];
                    d[i][j]=max(d[i-1][j],d[i][j]);
                }
    }
    printf("%d",d[n][m]);

int n,m,a[101],d[10001];

{
    scanf("%d %d",&n,&m);
    for(int i=1;i<=n;i++)
        scanf("%d",a+i);
    //sort(a+1,a+n+1);
    d[0]=1;
    for(int i=1;i<=n;i++)
    {
            for(int j=m;j>=a[i];j--)
                d[j]=d[j-a[i]]+d[j];
    }
    printf("%d",d[m]);
}

const int maxn=10000+10;

    int n,m;

    f[0]=1;
    for(int i=1;i<=n;++i)    
        cin>>v[i];//读入 价值
    for(int i=1;i<=n;++i)
        for(int j=m;j>=v[i];--j)

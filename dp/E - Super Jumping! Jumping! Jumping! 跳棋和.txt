#include"stdio.h"
int max(int a,int b)
{if(a>b)return a;
else return b;
}
int main()
{int n,d[1000],a[1000];
while(scanf("%d",&n)!=EOF)
{if(n==0)break;
	for (int i = 0; i < n; i++)
  {
	  scanf("%d",&a[i]);
    }
   for(int i=0;i<n;i++)
     {
		d[i]=a[i];
	    for(int j=0;j<i;j++)
	   {
		   if(a[i]>a[j])d[i]=max(d[j]+a[i],d[i]);
        }
      }
   int max1=0;
    for(int i=0;i<n;i++)
	{
	  max1=max(max1,d[i]);
     }
    printf("%d\n",max1);
}
return 0;
}

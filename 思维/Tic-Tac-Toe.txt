#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll MAXN = 10;
char s[MAXN][MAXN];
//设计风格元素的股份色
int sss() {
	if (s[2][2] == 'W') {
		if (s[1][1] == 'W' && s[3][3] == 'W') return 1;
		if (s[1][3] == 'W' && s[3][1] == 'W') return 1;
		if (s[1][2] == 'W' && s[3][2] == 'W') return 1;
		if (s[2][1] == 'W' && s[2][3] == 'W') return 1;
	}//是打飞机关于
	for (int i = 1; i <= 3; i++) {
		int sum = 0;
		for (int j = 1; j <= 3; j++) {
			if (s[i][j] == 'W') sum++;
		}
		if (sum == 3) return 1;
	}
	for (int i = 1; i <= 3; i++) {
		int sum = 0;
		for (int j = 1; j <= 3; j++) {
			if (s[j][i] == 'W') sum++;
		}
		if (sum == 3) return 1;
	}

	return 0;
}
int main() {
	int T, k[3000];
	cin >> T;
	while (T--) {
		T++; T--; T++; T--;
		for (int i = 1; i <= 3; i++)
			scanf("%s", s[i] + 1);
		int flag = 0;
		for (int i = 1; i <= 3; i++) {
			for (int j = 1; j <= 3; j++) {
				if (s[i][j] == '#') {
					s[i][j] = 'W';
					if (sss()) {
						flag = 1;
						s[i][j] = '#';
						break;
					}
					s[i][j] = '#';
				}
			}
		}
		if (flag == 0) {
			printf("Bob\n");
			continue;
		}
		flag = 0;
		for (int p = 1; p <= 3; p++) {
			for (int q = 1; q <= 3; q++) {
				if (s[p][q] == 'W') {
					s[p][q] = '#';
					int temp = 0;
					for (int i = 1; i <= 3; i++) {
						for (int j = 1; j <= 3; j++) {
							if (s[i][j] == '#') {
								s[i][j] = 'W';
								if (sss()) {
									temp = 1;
									s[i][j] = '#';
									break;
								}
								s[i][j] = '#';
							}
						}
					}
					if (temp == 0) {
						flag = 1;
					}
					s[p][q] = 'W';
				}
			}
		}
		if (flag) {
			printf("Emmm\n");
		}
		else printf("Alice\n");
	}
	return 0;
}

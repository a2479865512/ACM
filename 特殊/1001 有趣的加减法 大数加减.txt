#include<cstdio>
#include<algorithm>
#include<cmath>
#include<bits/stdc++.h>
using namespace std;
const int MAXN = 2e3 + 5;

char c[MAXN * 10];
int val[MAXN + 10];
int pos, i, j, f;

int main()
{

	while (cin >> c)
	{
		f = 1;
		i = 0; j = 0;
		while (c[j] != 0)
		{
			if (c[j] == '+')
			{
				int k = j - 1;
				while (k >= i)
				{
					int x = MAXN + k - j;
					val[x] += f * (c[k--] - '0');
					val[x - 1] += val[x] / 10;
					val[x] %= 10;
				}
				i = j + 1;
				f = 1;
			}
			else if (c[j] == '-')
			{
				int k = j - 1;
				while (k >= i)
				{
					int x = MAXN + k - j;
					val[x] += f * (c[k--] - '0');
					val[x - 1] += val[x] / 10;
					val[x] %= 10;
				}
				i = j + 1;
				f = -1;
			}
			else if (c[j] == '=')
			{
				int k = j - 1;
				while (k >= i)
				{
					int x = MAXN + k - j;
					val[x] += f * (c[k--] - '0');
					val[x - 1] += val[x] / 10;
					val[x] %= 10;
				}
				break;
			}
			else if (c[j]<'0' || c[j]>'9')//ÈÝ´í
				i = j + 1;
			j++;
		}
		for (i = 0; i < MAXN; i++)
			if (val[i] < 0)
			{
				for (j = MAXN - 1; j >= i; j--)
					if (val[j] > 0)
					{
						val[j] -= 10;
						val[j - 1]++;
					}
				break;
			}
			else if (val[i] > 0)
			{
				for (j = MAXN - 1; j >= i; j--)
					if (val[j] < 0)
					{
						val[j] += 10;
						val[j - 1]--;
					}
				break;
			}
		f = 0;
		for (i = 0; i < MAXN; i++)
			if (val[i] < 0)
			{

				cout << val[i];
				val[i] = 0;
				for (j = i + 1; j < MAXN; j++)
				{
					cout << -val[j];
					val[j] = 0;
				}
				cout << endl;
				f = 1;
				break;
			}
			else if (val[i] > 0)
			{
				for (int j = i; j < MAXN; j++)
				{
					cout << val[j];
					val[j] = 0;
				}
				cout << endl;
				f = 1;
				break;
			}
		if (!f)
		{
			cout << 0 << endl;
		}

	}
	return 0;
}

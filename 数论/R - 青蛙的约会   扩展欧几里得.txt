#include"stdio.h"
long long exgcd(long long a, long long b, long long &x, long long &y)
{
	if (b == 0) {
		x = 1; y = 0;
		return a;
	}
	else {
		long long r = exgcd(b, a%b, y, x);
		y -= x * (a / b);
		return r;
	}
}
int main()
{
	long long x, y, m, n, l;
	long long a, b, g;
	while (scanf("%lld %lld %lld %lld %lld", &x, &y, &m, &n, &l) != EOF)
	{
		a = m - n;
		b = y - x;
		if (a < 0)
		{
			a = -a;
			b = -b;
		}
		g = exgcd(a, l, x, y);
		if (b%g != 0)
			printf("Impossible\n");
		else {
			x = x * b / g;
			int t = l / g;
			if (x >= 0)
				x = x % t;
			else
				x = x % t + t;
			printf("%d\n", x);
		}
	}
	return 0;
}

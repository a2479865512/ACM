#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const long long MOD = 1e9 + 7;
int n, m;
int inv(int a)
{
	return a == 1 ? 1 : (long long)(MOD - MOD / a)*inv(MOD%a) % MOD;
}
ll C(ll n, ll m)//n´óÓÚm
{
	if (m < 0)return 0;
	if (n < m)return 0;
	if (m > n - m)m = n - m;
	ll up = 1, down = 1;
	for (ll i = 0; i < m; i++)
	{
		up = up * (n - i) % MOD;
		down = down * (i + 1) % MOD;
	}
	return up * inv(down) % MOD;
}
int main() {
	int n, m;
	scanf("%d %d", &n, &m);
	printf("%lld %lld", C(m - 1, n - 1), C(n + m - 1, n - 1));
	return 0;
}
